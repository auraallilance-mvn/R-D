Project Overview

This project demonstrates the transition from local hardware control to cloud-based management. By utilizing the Arduino IoT Cloud and an ESP32 microcontroller, we developed a system where a single toggle on a web dashboard triggers a physical LED response instantly.

The system showcases real-time bidirectional communication between cloud services and embedded devices, achieving sub-200ms latency for control commands. This forms the foundation for scalable home automation and industrial IoT applications.

Full Source Code

Complete Arduino implementation with cloud callback function:

#include "thingProperties.h"

#define LED_PIN 2  // Built-in LED on ESP32

void setup() {
  Serial.begin(115200);
  delay(1500);
  
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);
  
  // Initialize Arduino IoT Cloud
  initProperties();
  ArduinoCloud.begin(ArduinoIoTPreferredConnection);
  
  setDebugMessageLevel(2);
  ArduinoCloud.printDebugInfo();
  
  Serial.println("‚úÖ WiFi LED Control System Initialized");
}

void loop() {
  ArduinoCloud.update();
  // Cloud handles all logic via callback
}

/* 
 * This function is called automatically whenever 
 * the 'led_control' variable changes on the 
 * Arduino Cloud dashboard.
 */
void onLedControlChange() {
  if (led_control == 1) {
    digitalWrite(LED_PIN, HIGH);
    Serial.println("Action: LED ON via Cloud");
  } else {
    digitalWrite(LED_PIN, LOW);
    Serial.println("Action: LED OFF via Cloud");
  }
}
