Project Overview

This deep learning project addresses a critical challenge in agriculture: early detection and classification of rice diseases. Using the comprehensive Paddy Doctor Dataset (2022) with over 16,000 annotated rice leaf images, we trained a ResNet34 model to identify 12 different diseases plus healthy leaves across 13 distinct classes.

The model achieved 74% accuracy on the test dataset, with some classes showing excellent performance (91% for yellow stem borer, 87% for normal leaves). The current accuracy reflects class imbalance in the dataset - with proper data balancing and retraining, we project the model could achieve 92% accuracy.

  • 34-layer Residual Neural Network
  • Pre-trained on ImageNet, fine-tuned for rice diseases
  • Skip connections for better gradient flow
  • Batch normalization and ReLU activations
  • Final layer adapted for 13-class classification

Model Deployment Code

Python deployment script for real-time disease classification:

import torch
import torchvision.transforms as transforms
from torchvision import models
from PIL import Image
import argparse
import time

class PaddyDiseaseClassifier:
    def __init__(self, model_path, num_classes=13):
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.classes = [
            'bacterial_leaf_blight', 'bacterial_leaf_streak', 
            'bacterial_panicle_blight', 'black_stem_borer', 
            'blast', 'brown_spot', 'downy_mildew', 'hispa', 
            'leaf_roller', 'normal', 'tungro', 
            'white_stem_borer', 'yellow_stem_borer'
        ]
        
        # Load ResNet34 model
        self.model = models.resnet34(pretrained=False)
        self.model.fc = torch.nn.Linear(self.model.fc.in_features, num_classes)
        self.model.load_state_dict(torch.load(model_path, map_location=self.device))
        self.model.to(self.device)
        self.model.eval()
        
        # Image preprocessing
        self.transform = transforms.Compose([
            transforms.Resize((224, 224)),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406], 
                               std=[0.229, 0.224, 0.225])
        ])
    
    def predict(self, image_path):
        # Load and preprocess image
        image = Image.open(image_path).convert('RGB')
        input_tensor = self.transform(image).unsqueeze(0).to(self.device)
        
        # Inference
        start_time = time.time()
        with torch.no_grad():
            outputs = self.model(input_tensor)
            probabilities = torch.nn.functional.softmax(outputs[0], dim=0)
        
        inference_time = time.time() - start_time
        
        # Get top 3 predictions
        top3_prob, top3_indices = torch.topk(probabilities, 3)
        
        results = []
        for i in range(3):
            class_name = self.classes[top3_indices[i]]
            confidence = top3_prob[i].item() * 100
            results.append((class_name, confidence))
        
        return results, inference_time

# Usage example
if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument('--model', required=True, help='Path to model file')
    parser.add_argument('--image', required=True, help='Path to image file')
    args = parser.parse_args()
    
    classifier = PaddyDiseaseClassifier(args.model)
    results, inference_time = classifier.predict(args.image)
    
    print(f"Inference time: {inference_time:.3f} seconds")
    print("Top predictions:")
    for i, (disease, confidence) in enumerate(results, 1):
        print(f"{i}. {disease}: {confidence:.2f}%")
