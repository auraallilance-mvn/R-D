Project Overview

This seven segment display clock project marks my first step into the world of electronics and embedded programming. Under the guidance of Prof. M.V. Narayana Sir, we learned the fundamentals of digital displays, Arduino programming, and circuit design through this hands-on project.

The project displays time in HH:MM format using four seven-segment displays. It demonstrates basic concepts of multiplexing, digit control, and time management in embedded systems. Hardware Setup

Components Required

  • Arduino Uno Development Board
  • 4× Seven Segment Common Cathode Displays
  • 4× NPN Transistors (2N2222 or similar)
  • 8× Current Limiting Resistors (220Ω)
  • 4× Base Resistors (1kΩ)
  • Breadboard and jumper wires
  • Power supply (5V)

Circuit Connections

Seven Segment Pins (A-G, DP):

Arduino D2-D9 → Segment pins (through 220Ω resistors)

Digit Control (Common Cathodes):

Arduino D10 → Transistor 1 (Hours tens)

Arduino D11 → Transistor 2 (Hours units)

Arduino D12 → Transistor 3 (Minutes tens)

Arduino D13 → Transistor 4 (Minutes units)

Note: Use transistors for digit switching to handle current

Arduino Source Code

Complete Arduino code for the seven segment display clock:

// Seven Segment Display Clock
// First Electronics Learning Project
// Mentor: MV Narayana Sir

// Pin definitions for segments (A-G, DP)
int segmentPins[] = {2, 3, 4, 5, 6, 7, 8, 9};

// Pin definitions for digit control
int digitPins[] = {10, 11, 12, 13};

// Seven segment patterns for digits 0-9
byte digitPatterns[10] = {
  B11111100, // 0
  B01100000, // 1
  B11011010, // 2
  B11110010, // 3
  B01100110, // 4
  B10110110, // 5
  B10111110, // 6
  B11100000, // 7
  B11111110, // 8
  B11110110  // 9
};

// Time variables
int hours = 12;
int minutes = 30;
int seconds = 0;
unsigned long previousMillis = 0;
const long interval = 1000; // 1 second

// Display variables
int currentDigit = 0;
unsigned long displayTimer = 0;
const int displayDelay = 5; // 5ms per digit

void setup() {
  // Initialize segment pins as outputs
  for (int i = 0; i < 8; i++) {
    pinMode(segmentPins[i], OUTPUT);
  }
  
  // Initialize digit pins as outputs
  for (int i = 0; i < 4; i++) {
    pinMode(digitPins[i], OUTPUT);
    digitalWrite(digitPins[i], HIGH); // Turn off all digits initially
  }
  
  Serial.begin(9600);
  Serial.println("Seven Segment Clock Started");
}

void loop() {
  // Update time every second
  unsigned long currentMillis = millis();
  if (currentMillis - previousMillis >= interval) {
    previousMillis = currentMillis;
    updateTime();
  }
  
  // Display multiplexing
  if (currentMillis - displayTimer >= displayDelay) {
    displayTimer = currentMillis;
    displayDigit();
  }
}

void updateTime() {
  seconds++;
  if (seconds >= 60) {
    seconds = 0;
    minutes++;
    if (minutes >= 60) {
      minutes = 0;
      hours++;
      if (hours > 23) {
        hours = 0;
      }
    }
  }
  
  // Print time to serial monitor
  Serial.print(hours < 10 ? "0" : "");
  Serial.print(hours);
  Serial.print(":");
  Serial.print(minutes < 10 ? "0" : "");
  Serial.println(minutes);
}

void displayDigit() {
  // Turn off all digits
  for (int i = 0; i < 4; i++) {
    digitalWrite(digitPins[i], HIGH);
  }
  
  // Determine which digit to display
  int digitValue;
  switch (currentDigit) {
    case 0: // Hours tens
      digitValue = hours / 10;
      break;
    case 1: // Hours units
      digitValue = hours % 10;
      break;
    case 2: // Minutes tens
      digitValue = minutes / 10;
      break;
    case 3: // Minutes units
      digitValue = minutes % 10;
      break;
  }
  
  // Set segment pattern
  setSegments(digitPatterns[digitValue]);
  
  // Turn on current digit
  digitalWrite(digitPins[currentDigit], LOW);
  
  // Move to next digit
  currentDigit = (currentDigit + 1) % 4;
}

void setSegments(byte pattern) {
  for (int i = 0; i < 8; i++) {
    digitalWrite(segmentPins[i], (pattern >> (7-i)) & 1);
  }
}

// Function to set time manually (for testing)
void setTime(int h, int m, int s) {
  hours = h;
  minutes = m;
  seconds = s;
}
